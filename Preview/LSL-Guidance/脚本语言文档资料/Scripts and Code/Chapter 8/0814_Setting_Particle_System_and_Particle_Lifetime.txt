default
{
    touch_start(integer total_number)
    {
        // Create a brief explosion of some long-lived particles
        llParticleSystem( [
            PSYS_SRC_PATTERN, PSYS_SRC_PATTERN_EXPLODE,
            PSYS_SRC_MAX_AGE, 2.0,
            PSYS_PART_MAX_AGE, 10.0
            ]);
            
            llSetTimerEvent(2.0);
    }
    timer()
    {
        // Turn the timer event off
        llSetTimerEvent(0.0);
        
        // Turn the particles off
        llParticleSystem([]);
    }
}
