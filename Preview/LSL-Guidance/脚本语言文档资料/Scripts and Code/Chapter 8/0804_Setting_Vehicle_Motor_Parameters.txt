default
{
    state_entry()
    {
        llSetStatus(STATUS_PHYSICS, TRUE);
        llSetVehicleType(VEHICLE_TYPE_CAR);
        
        // The vehicle will get up to full speed in 1 second
        llSetVehicleFloatParam(VEHICLE_LINEAR_MOTOR_TIMESCALE, 1.0);
        
        // The motor will become ineffective after 3 seconds.
        llSetVehicleFloatParam(VEHICLE_LINEAR_MOTOR_DECAY_TIMESCALE, 
            3.0);
    }
    
    touch_start(integer num_detected)
    {
        // Turn on the motor
        // Set to motor to go 20 meters/second forward
        // Note that the motor timescale and motor decay timescale are
        // parameters of the vehicle, and have to be set only once. 
        llSetVehicleVectorParam(VEHICLE_LINEAR_MOTOR_DIRECTION, 
            <20.0, 0.0, 0.0>);
    }
}
