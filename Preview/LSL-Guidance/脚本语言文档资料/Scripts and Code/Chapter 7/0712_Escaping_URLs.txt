// URL to search digg.com, the search term is added to the end of this
string searchURLBase = "http://www.digg.com/search?section=news&s=";

searchDigg(string searchTerm)
{
    // build the URL
    string url = 
        searchURLBase + 
        llEscapeURL(searchTerm);
    llLoadURL(llGetOwner(), "", url);
}
 
default
{
    state_entry() 
    {
        // search for stories about "second life" on digg.com 
        searchDigg("\"second life\"");
    }
}
