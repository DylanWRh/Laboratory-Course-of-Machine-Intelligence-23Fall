default
{
    state_entry()
    {
        // Create a series of concentric expanding rings of particles
        llParticleSystem([
            PSYS_SRC_PATTERN, PSYS_SRC_PATTERN_ANGLE,
            
            // Emit particles in a complete circle
            PSYS_SRC_ANGLE_BEGIN, 0.0,
            PSYS_SRC_ANGLE_END, PI,
            
            // Make all the particles spread out at the same speed
            PSYS_SRC_BURST_SPEED_MIN, 0.3,
            PSYS_SRC_BURST_SPEED_MAX, 0.3,
            
            // Start the ring 1 meter from the emitter
            PSYS_SRC_BURST_RADIUS, 1.0,
            
            // Emit 50 particles every 3 seconds
            PSYS_SRC_BURST_PART_COUNT, 50,
            PSYS_SRC_BURST_RATE, 3.0
            ]);
    }
}
